Bei mir so: (IIRC aus dem Zimmers übernommen)

#v+
# Tastatur in der bash

#set editing-mode vi
#set keymap vi
set convert-meta off
set meta-flag on
set output-meta on

"\e[1~": beginning-of-line
"\e[3~": delete-char
"\e[4~": end-of-line
"\e[A": previous-history
"\e[B": next-history
"\e[D": backward-char
"\e[C": forward-char
"\"": self-insert
