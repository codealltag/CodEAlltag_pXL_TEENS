Anfang:
FOR patsch% = 1 TO 50
Zahl = RND(1)
IF Zahl <= .33 THEN PRINT "Wilhelm Zumdiek ist viel besser als LDC"
IF Zahl > .33 AND Zahl <= .66 THEN PRINT "Käte ist doof"
IF Zahl > .66 THEN PRINT "*PATSCH*"
SLEEP (1)
NEXT patsch%
Frage:
INPUT "Hast du's jetzt endlich kapiert? [j/n]",antwort$
IF antwort$ = "n" THEN GOTO Anfang
ELSEIF antwort$ = "j" THEN GOTO Schluss 
DAGMAR PRINT "Bitte nur j/n!"
ENDIF
GOTO Frage
